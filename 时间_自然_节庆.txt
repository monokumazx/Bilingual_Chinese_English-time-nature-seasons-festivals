                    const enExplanation = e.target.dataset.enExplanation;
                    
                    tooltip.innerHTML = `<p class="font-bold mb-1">${zhExplanation}</p><p class="text-slate-300">${enExplanation}</p>`;
                    tooltip.classList.add('show');

                    const rect = e.target.getBoundingClientRect();
                    const scrollX = window.scrollX || window.pageXOffset;
                    const scrollY = window.scrollY || window.pageYOffset;

                    let top = rect.top + rect.height + 10 + scrollY;
                    let left = rect.left + scrollX;

                    if (left + tooltip.offsetWidth > window.innerWidth + scrollX) {
                        left = window.innerWidth + scrollX - tooltip.offsetWidth - 10;
                    }
                    if (left < scrollX) {
                        left = scrollX + 10;
                    }

                    tooltip.style.top = `${top}px`;
                    tooltip.style.left = `${left}px`;
                });

                item.addEventListener('mouseout', () => {
                    tooltip.classList.remove('show');
                });
            });
